local RS = game:GetService("ReplicatedStorage")
local Lasso = require(RS._replicationFolder:WaitForChild("Lasso"))

-- Wrap doLasso
local oldDo = Lasso.doLasso
Lasso.doLasso = function(...)
    print("[Spy] doLasso called with:", ...)
    return oldDo(...)
end

-- Wrap _hit
local oldHit = Lasso._hit
Lasso._hit = function(...)
    print("[Spy] _hit called with:", ...)
    return oldHit(...)
end
